const express = require('express'); const multer = require('multer'); const path = require('path'); const fs = require('fs'); const app = express(); const PORT = process.env.PORT || 10000; // Create necessary folders if (!fs.existsSync('uploads')) { fs.mkdirSync('uploads'); console.log('Created uploads folder'); } if (!fs.existsSync('public')) { fs.mkdirSync('public'); console.log('Created public folder'); } // Create a simple index.html if it doesn't exist const indexPath = path.join(__dirname, 'public', 'index.html'); if (!fs.existsSync(indexPath)) { const simpleHtml = ` Image Host

Image Hosting Service

Upload an image to get a shareable link

`; fs.writeFileSync(indexPath, simpleHtml); console.log('Created default index.html'); } // Multer setup const storage = multer.diskStorage({ destination: function (req, file, cb) { cb(null, 'uploads/'); }, filename: function (req, file, cb) { const uniqueName = Date.now() + '-' + Math.round(Math.random() * 1E9) + path.extname(file.originalname); cb(null, uniqueName); } }); const upload = multer({ storage: storage, limits: { fileSize: 5 * 1024 * 1024 } // 5MB }); // Middleware app.use(express.static('public')); app.use('/uploads', express.static('uploads')); // Routes app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); app.post('/upload', upload.single('image'), (req, res) => { try { if (!req.file) { return res.status(400).json({ error: 'No file uploaded' }); } // Create the URL const protocol = req.headers['x-forwarded-proto'] || req.protocol; const host = req.get('host'); const imageUrl = `${protocol}://${host}/uploads/${req.file.filename}`; res.json({ success: true, message: 'Upload successful!', filename: req.file.filename, url: imageUrl, size: (req.file.size / 1024).toFixed(2) + ' KB' }); } catch (error) { res.status(500).json({ error: 'Upload failed: ' + error.message }); } }); // Health check app.get('/health', (req, res) => { res.json({ status: 'OK', timestamp: new Date() }); }); // Start server app.listen(PORT, () => { console.log(`âœ… Server running on port ${PORT}`); console.log(`ğŸ“ Uploads folder: ${__dirname}/uploads`); console.log(`ğŸ“ Public folder: ${__dirname}/public`); console.log(`ğŸŒ Open: http://localhost:${PORT}`); });
